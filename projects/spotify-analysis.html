%matplotlib inline
# Displaying the descriptive statistics
desc_stats = spotify_data.describe().T  # Transpose for better readability

# let's have a quick double check for missing values and data types as well in descriptive statistics 
desc_stats['missing_values'] = spotify_data.isnull().sum()
desc_stats['unique_values'] = spotify_data.nunique()
desc_stats['dtype'] = spotify_data.dtypes

# Rename index to a proper column name for better display
desc_stats.index.name = 'Feature'

# Display the descriptive statistics DataFrame
print(desc_stats)

# lets also visualize the descriptive stats
# Select only numeric columns for visualization as histogram and distribution
num_columns = [
    'duration_ms', 'year', 'acousticness', 'danceability', 'energy',
    'instrumentalness', 'liveness', 'loudness', 'speechiness', 'tempo',
    'valence', 'popularity'
]

# Set up the figure size
plt.figure(figsize=(15, 12))

# Loop through columns and create histograms
for i, col in enumerate(num_columns, 1):
    plt.subplot(4, 4, i)  # Create a grid of subplots
    sns.histplot(spotify_data[col], bins=30, kde=True)
    plt.title(f'Distribution of {col}')
    
plt.tight_layout()  # Adjust layout for better spacing
plt.show()

# boxplots and checking for outliers
plt.figure(figsize=(15, 10))

for i, col in enumerate(num_columns, 1):
    plt.subplot(4, 4, i)
    sns.boxplot(x=spotify_data[col])
    plt.title(f'Box Plot of {col}')

plt.tight_layout()
plt.show()
